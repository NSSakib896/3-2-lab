.model small
.stack 100h
 
.code
 
main proc
    ;input 
    mov ah,1
    int 21h
    mov cl,al 
 
    ;newline
    mov ah,2
    mov dl,10
    int 21h
    mov dl,13
    int 21h
 
    ;loop1 start
    mov ch,'0'
    l1:      
        inc ch   
        ;loop2 start
        mov dh,'0'     
        l2:
            inc dh
 
            mov ah,2    ;print 1
            mov dl,'1'
            int 21h
 
            cmp dh,ch
            jl l2
 
 
        mov ah,2     ;newline
        mov dl,10
        int 21h
        mov dl,13
        int 21h
 
        cmp ch,cl
        jl l1
    ;loop1 end
    
  exit:
  mov ah,4ch
  int 21h
  main endp
end main
